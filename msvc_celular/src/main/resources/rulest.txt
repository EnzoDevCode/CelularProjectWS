# Regla 1: Si un celular es de la marca Apple, entonces usa iOS
[AppleUsesIOS: 
    (?x rdf:type <http://schema.org/Celular>), 
    (?x <http://schema.org/brand> "Apple") 
    -> 
    (?x <http://schema.org/operatingSystem> "iOS")
]

# Regla 2: Si un celular es de la marca Samsung, entonces usa Android
[SamsungUsesAndroid: 
    (?x rdf:type <http://schema.org/Celular>), 
    (?x <http://schema.org/brand> "Samsung") 
    -> 
    (?x <http://schema.org/operatingSystem> "Android")
]
